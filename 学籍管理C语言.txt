#include <stdio.h>#include <string.h>#include <Windows.h>int sco[7];struct Node {	char id[20];	char name[20];	char dep[20];	double cou[6];	double sum;}tea[1000006];int op;int tot = 0;int i, j;int f;char name[100];typedef struct Node node;void Swap(node* a, node* b) {	node x;	strcpy((x).id, a->id);	strcpy((x).name, a->name);	strcpy((x).dep, a->dep);	for (int i = 0; i < 6; i++) {		x.cou[i] = a->cou[i];	}	///////////////////////////////////////	strcpy(a->id, b->id);	strcpy(a->name, b->name);	strcpy(a->dep, b->dep);	for (int i = 0; i < 6; i++) {		a->cou[i] = b->cou[i];	}	//////////////////////////////////////	strcpy(b->id, x.id);	strcpy(b->name, x.name);	strcpy(b->dep, x.dep);	for (int i = 0; i < 6; i++) {		b->cou[i] = x.cou[i];	}}void add() {	tot++;	printf("输入学生id: ");	scanf("%s", tea[tot].id);	printf("输入姓名: ");	scanf("%s", tea[tot].name);	printf("输入专业: ");	scanf("%s", tea[tot].dep);	printf("请依次输入程序设计基础，英语，高数，制图，物理，体育 6门成绩\n");	int tt = 0;	tea[tot].sum = 0;	for (i = 0; i < 6; i++) {		scanf("%lf", &tea[tot].cou[i]);		tea[tot].sum += (sco[i] * tea[tot].cou[i]);		tt += sco[i];	}	tea[tot].sum /= tt;}void show() {	for (i = 1; i <= tot; i++) {		printf("学生id： "); printf("%s\n", tea[i].id);		printf("姓名： "); printf("%s\n", tea[i].name);		printf("专业： "); printf("%s\n", tea[i].dep);		printf("程序设计基础\t英语\t高数\t制图\t物理\t体育\n");		for (j = 0; j < 6; j++)			printf("%.2lf\t", tea[i].cou[j]);		printf("\n总绩点：%.2lf\n", tea[i].sum);		printf("\n");	}}void find() {	f = 0;	printf("请输入要查找的学生姓名: "); scanf("%s", name);	for (i = 1; i <= tot; i++) {		if (strcmp(tea[i].name, name) == 0) {			f = 1;			printf("学生号： "); printf("%s\n", tea[i].id);			printf("姓名： "); printf("%s\n", tea[i].name);			printf("专业： "); printf("%s\n", tea[i].dep);			printf("程序设计基础\t英语\t高数\t制图\t物理\t体育\n");			for (j = 0; j < 6; j++)				printf("%.2lf\t", tea[i].cou[j]);			printf("\n总绩点：%.2lf\n", tea[i].sum);			printf("\n");			break;		}	}	if (!f) printf("未找到信息\n");}void del() {	f = 0;	printf("请输入要删除的学生姓名: ");	scanf("%s", name);	for (i = 1; i <= tot; i++) {		if (strcmp(tea[i].name, name) == 0) {			f = 1;			for (j = i + 1; j <= tot; j++) {				Swap(&tea[j - 1], &tea[j]);			}			tot--;		}	}	if (!f) printf("未找到信息\n");}void change() {	f = 0;	printf("请输入要修改的学生姓名: ");	scanf("%s", name);	for (i = 1; i <= tot; i++) {		if (strcmp(tea[i].name, name) == 0) {			f = 1;			printf("请依次输入程序设计基础，英语，高数，制图，物理，体育 6门成绩\n");			int tt = 0;			tea[tot].sum = 0;			for (i = 0; i < 6; i++) {				scanf("%lf", &tea[tot].cou[i]);				tea[tot].sum += (sco[i] * tea[tot].cou[i]);				tt += sco[i];			}			break;		}	}	if (!f) printf("未找到信息\n");}int main(){	printf("请依次输入程序设计基础，英语，高数，制图，物理，体育 6门学分\n");	for (int i = 0; i < 6; i++) {		scanf("%d", &sco[i]);	}	while (1) {		printf("1: 学生信息查询\n");		printf("2: 增加学生信息\n");		printf("3: 删除一名学生\n");		printf("4: 修改成绩\n");		printf("5: 排序\n");		printf("6: 打印\n");		printf("7: 成绩统计\n");		printf("8: 保存并退出\n");		scanf("%d", &op);		if (op == 1) {			find();		}		if (op == 2) {			add();		}		if (op == 3) {			del();		}		if (op == 4) {			change();		}		if (op == 5) {			printf("选择A,D分别表示升序和降序");			getchar();			char c; scanf("%c", &c);			printf("输入的数字标识(0,5)，对对应的成绩排序");			int tt; scanf("%d", &tt);			if (c == 'A') {				for (i = 1; i <= tot; i++) {					for (j = i + 1; j <= tot; j++) {						if (tea[i].cou[tt] > tea[j].cou[tt])							Swap(&tea[i], &tea[j]);					}				}			}			else {				for (i = 1; i <= tot; i++) {					for (j = i + 1; j <= tot; j++) {						if (tea[i].cou[tt] < tea[j].cou[tt])							Swap(&tea[i], &tea[j]);					}				}			}			show();		}		if (op == 6) {			show();		}		if (op == 7) {			/*			* 输入1-7，分别表示统计并显示不同课程的成绩。统计的信息包括：平均分，各等级多少人，分别占多少百分比。如90-100占比，80-89占比，，。。60分以下。			*/			printf("输入的数字标识(0,5)，对对应的课程统计");			int tt; scanf("%d", &tt);			double a, b, c, d, e;			double sum = 0;			a = b = c = d = e = 0;			for (int i = 1; i <= tot; i++) {				sum += tea[i].cou[tt];				if (tea[i].cou[tt] >= 90) a++;				else if (tea[i].cou[tt] >= 80) b++;				else if (tea[i].cou[tt] >= 70) c++;				else if (tea[i].cou[tt] >= 60) d++;				else e++;			}			printf("平均分: %.2lf\n>= 90: %.2lf\n>= 80: %.2lf\n>= 70: %.2lf\n>= 60: %.2lf\n<60:%.2lf\n", sum / tot, a / tot, b / tot, c / tot, d / tot, e / tot);		}		if (op == 8) {			FILE* file;			file = fopen("student.txt", "w");			for (i = 1; i <= tot; i++) {				fprintf(file, "%s\t%s\t%s\t%lf\t%lf\t%lf\t%lf\t%lf\t%lf\n", tea[i].id, tea[i].name, tea[i].dep,					tea[i].cou[0],					tea[i].cou[1],					tea[i].cou[2],					tea[i].cou[3],					tea[i].cou[4],					tea[i].cou[5],					tea[i].cou[6]);			}			fclose(file);			file = fopen("course.txt", "w");			fprintf(file, "%d\t%d\t%d\t%d\t%d\t%d", sco[0], sco[1], sco[2], sco[3], sco[4], sco[5]);			fclose(file);			break;		}		system("pause");		system("cls");	}	return 0;}
