class Men {
public:
	string id;
	string name;
	string sex;
	string call;
	string part;
	virtual void add() = 0;
	virtual void show() = 0;
	virtual void change() = 0;
	// 编号 姓名 性别 电话 专业
};

class Teacher :public Men {
public:
	void add() {
		// 编号 姓名 性别 电话 专业
		cout << "输入编号: "; cin >> id;
		cout << "输入姓名: "; cin >> name;

		cout << "输入电话: "; cin >> call;
		cout << "输入专业: "; cin >> part;
	}
	void show() {
		cout << "编号: " << id << endl;
		cout << "姓名: " << name << endl;
		cout << "性别: " << sex << endl;
		cout << "电话: " << call << endl;
		cout << "专业: " << part << endl;
	}
	void change() {
		while (1) {
			system("cls");
			printf("0: 退出\n");
			printf("1: 修改编号\n");
			printf("2: 修改姓名\n");
			printf("3: 修改性别\n");
			printf("4: 修改电话\n");
			printf("5: 修改专业\n");
			int op; cin >> op;

			if (!op) break;
			if (op == 1) {
				cout << "输入编号: "; cin >> id;
			}
			if (op == 2) {
				cout << "输入姓名: "; cin >> name;
			}
			if (op == 3) {
				cout << "输入性别: "; cin >> sex;
			}
			if (op == 4) {
				cout << "输入电话: "; cin >> call;
			}
			if (op == 5) {
				cout << "输入专业: "; cin >> part;
			}
			system("pause");
		}
	}
};
